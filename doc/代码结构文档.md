1.公司项目后台架构
``` lua
bigdata
├── config -- 配置 除了新加配置或者修改数据源配置基本不用动
|    ├── aspect -- 切面操作 目前的功能：service接口统计
|    ├── interceptor -- 拦截器
|    ├── listener -- 监听器
|    └── security -- 权限处理
├── controller -- 控制分发器
|    ├── AlarmController.java -- 预警相关业务类
|    ├── BehaviorController.java -- 画像相关业务类
|    ├── PortraitController.java -- 行为轨迹相关业务类
|    ├── ReportController.java -- 大学生报告业务类
|    └── ... -- 增加其他的相关业务类
├── dao -- 数据操作层
|    ├── alarm -- 预警相关dao业务类
|    ├── behavior -- 画像相关dao业务类
|    ├── portrait -- 行为轨迹相关dao业务类
|    ├── report -- 学生报告相关dao业务类
|    └── ... -- 增加其他的相关业务类
├── model -- 实体类模型层
|    ├── entity -- 数据库对应的表模型
|    ├── enums -- 枚举
|    ├── req -- 入参模型
|    └── res -- 出参模型
├── service -- 业务
|    ├── impl -- 业务实现
|    |    ├── alarm -- 预警相关实现业务类
|    |    ├── behavior -- 画像相关实现业务
|    |    ├── portrait -- 行为轨迹相关实现业务
|    |    ├── report -- 学生报告相关实现业务
|    |    └── ... -- 其他实现业务
|    ├── alarm -- 预警相关接口业务
|    ├── behavior -- 画像相关接口业务
|    ├── portrait -- 行为轨迹相关接口业务
|    ├── report -- 学生报告相关接口业务
|    └── ... -- 其他接口业务
└──  BigDataApplication -- 启动main类[默认端口:8080]


resources
└──   mybatis -- 数据库操作
|    ├── alarm -- 预警相关dao实现业务类
|    ├── behavior -- 画像相关dao实现业务类
|    ├── portrait -- 行为轨迹相关dao实现业务类
|    ├── report -- 学生报告相关dao实现业务类
|    └── ... -- 增加其他的实现业务类

   bigdata-web -- 代码架构
```
